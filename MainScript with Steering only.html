%%%Script of the full vehicle with Braking and Steering systems
clearvars
clc
 
%% Simulation settings
%initial velocity in km/h
vi = 80;
%Driving system Switch
Driving = false;
%Braking system Switch
Braking = false;
%Steering system Switch
Steering = true;
%Suspension Geometry Switch
Sus_switch = false;
%Aerodynamics Drag Switch
Aero_switch = true;
%Rolling Resistance Switch
RR_switch = true;
%Road Profile Switch
Road_switch = false;
Switches = [Driving, Braking, Steering, Sus_switch, Aero_switch, RR_switch, Road_switch];
 
 
%%Importing System Parameters
[Tire, Ve, St, Br, Dr] = VehParameters();
%{
tire: tire parameters struct, Ve: vehicle parameters struct,
St: steering system parameters struct, Br: braking system parameters struct.
Dr: driving system parameters struct
%}
[p1, p2, p3, p4] = sus_generation(Ve); %importing Suspension Geomerty hardpoints
[sss,~] = dwb(0,0,p1);
St.iSL = abs(sss.i_SL);
v0 = vi/3.6; % the intial velocity of the vehicle [m/s]
 
%%simulation time
t = 0; % the start time of simulation [s].
tf = 30; % the end time of simulation [s].
tstep = 3*10^-4; % the time step [s].
iout = 2;
N = round ( tf /( iout * tstep ))+1; % the number of iterations
%%Inputs of the scenario
 
%%Pressure set point in bar
pressure_set = 15*Braking; % in [bar]
%{
pc1_set = @(t) (0) *(t <= 0.2) + pressure_set*10^(5).*(t > 0.2 && t <=1.3) + ...
(pressure_set*10^(5)-pressure_set*10^(5).*(t-1.3)/0.5).*(t > 1.3 && t<=1.8) + 0.*(t > 1.8 && t<= 2.5)...
+pressure_set*10^(5).*(t > 2.5 && t <=3.7) + (pressure_set*10^(5)-pressure_set*10^(5).*(t-3.7)/0.5)...
.*(t > 3.7 && t<=4.2) + 0*(t > 4.2); % % step input
%}
pc1_set = @(t) (0) *(t <= 0.2) + pressure_set*10^(5).*(t > 0.2) ; % % step input
 
ts = 0:0.01:tf;
pc1set = zeros(length(ts), 1);
for i = 1: length(ts)
pc1set(i) = pc1_set(ts(i));
end
%%Steering set point in rad
%{
degree1 = 20*Steering; % to know if there is steering or not
degree2 = 38*Steering;
Tw1 = 2;
Tw2 = 2.5;
angle_setpoint = @(t) (0)*(t<=0.1) + deg2rad(degree1)*sin(2*pi()*(t-0.1)/Tw1)*(t >0.1 && t <=(0.1+Tw1))...
+ 0*(t > (0.1+Tw1) && t <= (0.2+Tw1)) - deg2rad(degree2)*sin(2*pi()*(t-(0.2+Tw1))/Tw2)...
*(t >(0.2+Tw1) && t <=(0.2+Tw1+Tw2)) + 0*(t > (0.2+Tw1+Tw2));
%}
angle_setpoint = @(t) (0)*(t<=1)*Steering + deg2rad(3)*((t-1)/0.3)*(t>1&&t<1+0.3)*Steering+...
deg2rad(3)*(t>=1+0.3)*Steering;
angleset = zeros(length(ts), 1);
for i = 1: length(ts)
angleset(i) = angle_setpoint(ts(i));
end
 
%%Driving set point in throttle percentage
tqd_t = tf*...
[0.0 0.1 0.15 0.4 0.5 1]; % time for driving torque
Th_bedal = 0.98; % throttle value shouldn't be more than 1
tqd_v = Th_bedal*Driving* ...
[ 0.0 0.00 1.00 1.00 0.00 0.00 ];
throttle_setpoint = @(t) interp1(tqd_t, tqd_v, t, "linear", "extrap"); % Valid Expression
throttleset = zeros(length(ts), 1);
for i = 1: length(ts)
throttleset(i) = throttle_setpoint(ts(i));
end
%%road profile is in VEHICLE MODEL
 
%%intial conditions
x = zeros(46,1); % intitial conditions
x(4) = v0;
x(21) = v0/Ve.Radius;
x(22) = v0/Ve.Radius;
x(23) = v0/Ve.Radius;
x(24) = v0/Ve.Radius;
 
% pre-allocation
T = zeros (N ,1); % time matrix
X = zeros (N , length ( x )); % states matrix
Tb_p = zeros (N ,1); % brake torque at primary wheels [Nm]
Tb_s = zeros (N ,1); % brake torque at secondary wheels [Nm]
velxdot = zeros (N ,1); % the change in magnitude of the velocity component vx [m/s^2]
velydot = zeros (N ,1); % the change in magnitude of the velocity component vy [m/s^2]
ax = zeros (N ,1); % longitudinal acceleration of the vehicle [m/s^2]
ay = zeros (N ,1); % lateral acceleration of the vehicle [m/s^2]
sideslip = zeros (N,1); % side slip angle of the vehicle [rad]
k = zeros (N ,4); % the slip ratio of the front and rear wheels
fz = zeros (N ,4); % the normal load on the front and rear wheels [N]
fx = zeros (N ,4); % the longitudinal force on the front and rear wheels [N]
SlipAngle = zeros (N ,4); % the slip angle [rad]
fy = zeros (N ,4); % the lateral force on one tire [N]
Deltaf = zeros (N ,2); % The angels of the steering wheels [rad]
Thetaf = zeros (N ,1); % the common angel of the steering wheels [rad]
tp = zeros (N ,2); % the pneumatic trail [m]
U_B = zeros(N,1); % Control signal Braking [A]
U_S = zeros(N,1); % Control signal steering [v]
omegadot = zeros(N,4);
TD = zeros(N,4);
TB = zeros(N,4);
TRR = zeros(N,4);
 
out.theta_f = 0; %initialization of the struct "out"'s theta_f used in error eqn.
 
Brake_error_prior = 0; % Error at (k-1) time step
Brake_integral = 0; % Integral of error Braking
Steer_error_prior = 0; % Error at (k-1) time step for Steering
Steer_integral = 0; % Integral of error for Steering
%%solving the system differential equations by Heun - solver
for io = 1 : N
% Define desired setpoints/inputs
Brake_setpoint = pc1_set(t);
Steer_setpoint = angle_setpoint(t);
throttle_input = throttle_setpoint(t); %we can make it throttle by wire later
 
% Define PID Paramaters
[Brake_kp, Brake_ki, Brake_kd] = lookupBrake_Func(Brake_setpoint);
% Brake_kp = 3*10^(-5);
% Brake_ki = 0;
% Brake_kd = 0;
%[Steer_kp, Steer_ki, Steer_kd] = steerpid(x(4), rad2deg(Steer_setpoint));
Steer_kp = 2500;
Steer_ki = 500;
Steer_kd = 0;
% Simulate systems response with PID control
%Brake%
Brake_error = Brake_setpoint - x(29); % in [Pa]
Brake_integral = Brake_integral + Brake_error * tstep;
Brake_derivative = (Brake_error - Brake_error_prior) / tstep;
% Compute control signal for Braking%
U_B(io) = Brake_kp * Brake_error + Brake_ki * Brake_integral + Brake_kd * ...
Brake_derivative;
%Saturation, for braking (-50,50)%
if U_B(io) >=0
U_B(io)=min(U_B(io),24);
end
if U_B(io) <0
U_B(io)=max(U_B(io),-24);
end
% Update brake error for the next iteration
Brake_error_prior = Brake_error;
%Steer%
Steer_error = Steer_setpoint - out.theta_f;
Steer_integral = Steer_integral + Steer_error * tstep;
Steer_derivative = (Steer_error - Steer_error_prior) / tstep;
% Compute control signal for Steering%
U_S(io) = Steer_kp * Steer_error + Steer_ki * Steer_integral + Steer_kd * ...
Steer_derivative;
% Saturation value for the steering motor voltage (-12,12)
if U_S(io)>12
U_S(io)=12;
elseif U_S(io)<-12
U_S(io)=-12;
end
Steer_error_prior = Steer_error;
 
Inputs = [throttle_input; Steer_setpoint; Brake_setpoint];
PIDs = [U_S(io); U_B(io)];
[dx_dt, out] = vehModel( x, Tire, Ve, St, Br, Dr, p1, p2, p3, p4, Inputs, PIDs, Switches);
[x, dx_dt, PIDs] = ModelLogic( x, dx_dt, PIDs, Br);
 
%From generalscript%
%% generate output
T( io ) = t ; % the simulation time [s]
X(io ,:) = x.'; % the states
k(io,:) = out.k.'; % the slip ratio of the front and rear wheels
fz(io,:) = out.fz.'; % the normal load on the front and rear wheels [N]
fx(io,:) = out.fx.'; % the longitudinal force on the front and rear wheels [N]
fy(io,:) = out.fy.'; % the lateral force on one tire [N]
velxdot(io) = out.velxdot; % the change in magnitude of the velocity component vx [m/s^2]
velydot(io) = out.velydot; % the change in magnitude of the velocity component vy [m/s^2]
ax(io,1) = out.ax; % longitudinal acceleration of the vehicle [m/s^2]
ay(io,1) = out.ay; % lateral acceleration of the vehicle [m/s^2]
sideslip(io,1) = out.sideslip; % side slip abgle of the vehicle [rad]
Thetaf(io) = out.theta_f; % the commen angel of the steering wheels [rad]
Deltaf(io,:) = out.deltaf.'; % the angels of the steering wheels [rad]
SlipAngle(io,:) = out.SlipAngle.'; % the slip angle [rad]
tp(io,:) =out.tp.'; % the pneumatic trail [m]
Tb_p(io,1) = out.Tb_p; % brake torque at primary wheel [Nm]
Tb_s(io,1) = out.Tb_s; % brake torque at secondary wheel [Nm]
omegadot(io,:) = out.omegadot.';
TD(io,:) = out.TD.';
TB(io,:) = out.TD.';
TRR(io,:) = out.TRR.';
%% Huen Solver
if io < N
for i =1: iout
dx_dt1 = vehModel( x, Tire, Ve, St, Br, Dr, p1, p2, p3, p4, Inputs, PIDs, Switches);
t = t + tstep ;
dx_dt2 = vehModel( x + tstep*dx_dt1, Tire, Ve, St, Br, Dr, p1, p2, p3, p4, Inputs, PIDs, Switches);
dx_dt = 0.5*( dx_dt1 + dx_dt2 );
[x, dx_dt, ~] = ModelLogic( x, dx_dt, PIDs, Br);
x = x + tstep*dx_dt ;
end
%{
if (x(4) < 0)
x(4) = 0;
end
if (x(21) < 0)
x(21) =0;
end
if (x(22) < 0)
x(22) = 0;
end
if (x(23) < 0)
x(23) = 0;
end
if (x(24) < 0)
x(24) = 0;
end
if x(25) < 0
x(25) = 0;
x(34) = 0;
end
%}
end
 
end
